import requests  
from bs4 import BeautifulSoup  
import pandas as pd  
# Base URL of the site  
BASE_URL = "http://books.toscrape.com/catalogue/page-{}.html"  
# Lists to store the data  
titles = []  
prices = []  
ratings = []  
availability = []  
# Function to convert rating text to number  
def rating_to_number(rating_text):  
    mapping = {  
        "One": 1,  
        "Two": 2,  
        "Three": 3,  
        "Four": 4,  
        "Five": 5  
    }  
    return mapping.get(rating_text, 0)  
# Loop through pages (there are 50 pages)  
for page in range(1, 51):  
    url = BASE_URL.format(page)  
    response = requests.get(url)  
    soup = BeautifulSoup(response.text, "html.parser")  
    books = soup.find_all("article", class_="product_pod")  
    for book in books:  
        title = book.h3.a['title']  
        price = book.find("p", class_="price_color").text  
        rating_class = book.find("p", class_="star-rating")["class"][1]  
        rating = rating_to_number(rating_class)  
        avail = book.find("p", class_="instock  
availability").text.strip()   
        titles.append(title)  
        prices.append(price)  
        ratings.append(rating)  
        availability.append(avail) 
# Create a DataFrame  
data = pd.DataFrame({  
    "Title": titles,  
    "Price": prices,  
    "Rating": ratings,  
    "Availability": availability  
})  
# Save to CSV  
data.to_csv("books.csv", index=False)  
print("CSV file 'books.csv' has been created successfully!")  
 
import requests  
from bs4 import BeautifulSoup  
import pandas as pd  
url = 'http://books.toscrape.com/'  
books = []  
while url:  
    r = requests.get(url)  
    soup = BeautifulSoup(r.text, 'html.parser')  
    for book in soup.select('article.product_pod'):  
        title = book.h3.a['title']  
        price = book.select_one('p.price_color').text  
        rating = book.p['class'][1]  
        books.append({'Title': title, 'Price': price, 'Rating': rating})  
    next_btn = soup.select_one('li.next a')  
    url = 'http://books.toscrape.com/' + next_btn['href'] if next_btn  
else None  
df = pd.DataFrame(books)  
df.to_csv('books.csv', index=False)  
df.head()  
 
import requests  
from bs4 import BeautifulSoup  
import pandas as pd  
url = 'http://quotes.toscrape.com/'  
quotes_data = []  
while url:  
    r = requests.get(url)  
    soup = BeautifulSoup(r.text, 'html.parser')  
    for quote in soup.select('div.quote'):  
        text = quote.span.text  
        author = quote.small.text  
        tags = [t.text for t in quote.select('.tags a.tag')]  
        quotes_data.append({'Quote': text, 'Author': author, 'Tags': ',  
'.join(tags)})  
    next_btn = soup.select_one('li.next a')  
    url = 'http://quotes.toscrape.com' + next_btn['href'] if next_btn  
else None  
df = pd.DataFrame(quotes_data)  
df.to_csv('quotes.csv', index=False)  
df.head()  

import requests

commerce/static/computers/laptops'
